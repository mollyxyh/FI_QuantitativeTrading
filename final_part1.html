def mean_r(mu, var, t, dist = 'Q'): 
    m = r0+mu*t
    if (dist == 'F'):
        m -= var * pow(t,2)
    return m

#int{r(u)du}:
def mean_int_r(mu, t):

def var_int_r(var, t):

def A(mu, var, tau):
    return var*pow(tau,3)/6. - mu*pow(tau,2)/2.

def B(tau):
    return tau

def zero(mu, var, t0, T, r_t0):
    tau = T-t0
    return np.exp(A(mu, var, tau)-B(tau)*r_t0)

def fwd_rate(mu, var, t_fix, T0, T1, dist = 'Q', acrual=0.25):
    A_diff = A(mu, var, T0-t_fix) - A(mu, var, T1-t_fix)
    B_diff = B(T0-t_fix) - B(T1-t_fix)
    m = A_diff - B_diff * mean_r(mu, var, t_fix, dist)
    v = var * t_fix * pow(B_diff, 2)
    e = np.exp(m+v/2.)
    return 1/acrual * (e-1)

def mgf(mu, var, t_fix,T0,T1,k,dist = 'Q'):
    A_diff = A(mu, var, T0-t_fix) - A(mu, var, T1-t_fix)
    B_diff = B(T0-t_fix) - B(T1-t_fix)
    m = A_diff - B_diff * mean_r(mu, var, t_fix,dist)
    v = var * t_fix * pow(B_diff,2)
    return np.exp(k*(2*m+k*v)/2.)

def fwd_rate_mean(mu, var, t_fix, T0, T1, dist='Q', acrual=0.25):
    m1 = mgf(mu, var, t_fix, T0, T1, 1, dist)
    return 1/acrual*(m1-1)

def fwd_rate_vol(mu, var, t_fix, T0, T1, dist='Q', acrual=0.25):
    m2 = mgf(mu, var, t_fix, T0, T1, 2, dist)
    m1 = mgf(mu, var, t_fix, T0, T1, 1, dist)
    e2 = m2 - pow(m1, 2)
    return 1/acrual*pow(e2/t_fix,0.5)
